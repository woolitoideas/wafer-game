    <script>

        const industries = [

            "CPU", "GPU", "DRAM", "Flash", "CMOS", "Power",

            "Auto", "AI Core", "MEMS", "RF", "PMIC", "Panel",

            "FPGA", "ASIC", "SoC", "HBM", "ISP", "NPU"

        ];


        let currentLevel = 1;

        let timer = 0;

        let timerInterval;

        let flippedCards = [];

        let matchedCount = 0;

        let totalPairs = 0;

        let isProcessing = false;


        const boardConfigs = {

            1: { rows: 4, cols: 4 },

            2: { rows: 4, cols: 6 },

            3: { rows: 6, cols: 6 }

        };


        const board = document.getElementById('game-board');

        const levelDisplay = document.getElementById('level-display');

        const timerDisplay = document.getElementById('timer-display');

        const modal = document.getElementById('modal');

        const modalTitle = document.getElementById('modal-title');

        const modalContent = document.getElementById('modal-content');

        const modalBtn = document.getElementById('modal-btn');


        function initGame() {

            const config = boardConfigs[currentLevel];

            const totalCards = config.rows * config.cols;

            totalPairs = totalCards / 2;

            matchedCount = 0;

            flippedCards = [];

            isProcessing = false;


            let levelIndustries = [...industries].slice(0, totalPairs);

            let gameSet = [...levelIndustries, ...levelIndustries];

            gameSet.sort(() => Math.random() - 0.5);


            levelDisplay.textContent = `${currentLevel} / 3`;

            board.innerHTML = '';

            board.style.gridTemplateColumns = `repeat(${config.cols}, 1fr)`;

            board.style.gridTemplateRows = `repeat(${config.rows}, 1fr)`;


            gameSet.forEach((industry) => {

                const card = document.createElement('div');

                card.className = 'card';

                card.dataset.name = industry;

                

                card.innerHTML = `

                    <div class="card-face card-front"></div>

                    <div class="card-face card-back"><span>${industry}</span></div>

                `;


                card.addEventListener('click', () => flipCard(card));

                board.appendChild(card);

            });


            if (currentLevel === 1 && timer === 0) {

                startTimer();

            }

        }


        function flipCard(card) {

            if (isProcessing || card.classList.contains('flipped') || card.classList.contains('matched')) {

                return;

            }


            card.classList.add('flipped');

            flippedCards.push(card);


            if (flippedCards.length === 2) {

                checkMatch();

            }

        }


        function checkMatch() {

            isProcessing = true;

            const [card1, card2] = flippedCards;


            if (card1.dataset.name === card2.dataset.name) {

                setTimeout(() => {

                    card1.classList.add('matched');

                    card2.classList.add('matched');

                    matchedCount++;

                    flippedCards = [];

                    isProcessing = false;


                    if (matchedCount === totalPairs) {

                        handleLevelComplete();

                    }

                }, 400);

            } else {

                setTimeout(() => {

                    card1.classList.remove('flipped');

                    card2.classList.remove('flipped');

                    flippedCards = [];

                    isProcessing = false;

                }, 800);

            }

        }


        function handleLevelComplete() {

            if (currentLevel < 3) {

                showModal("製程區域完成！", `累計工時：${formatTime(timer)}`, "進入下一關");

            } else {

                clearInterval(timerInterval);

                showModal("產線完工！", `恭喜達成 100% 良率！總工時：${formatTime(timer)}`, "重新開始挑戰");

            }

        }


        function showModal(title, content, btnText) {

            modalTitle.textContent = title;

            modalContent.textContent = content;

            modalBtn.textContent = btnText;

            modal.classList.add('active');

        }


        modalBtn.addEventListener('click', () => {

            modal.classList.remove('active');

            if (currentLevel < 3) {

                currentLevel++;

                initGame();

            } else {

                currentLevel = 1;

                timer = 0;

                initGame();

                startTimer();

            }

        });


        function startTimer() {

            clearInterval(timerInterval);

            timerInterval = setInterval(() => {

                timer++;

                timerDisplay.textContent = formatTime(timer);

            }, 1000);

        }


        function formatTime(seconds) {

            const mins = Math.floor(seconds / 60);

            const secs = seconds % 60;

            return `${mins.toString().padStart(2, '0')}:${secs.toString().padStart(2, '0')}`;

        }


        window.onload = initGame;
